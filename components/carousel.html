const [offset, setOffset] = React.useState(0);
const [direction, setDirection] = React.useState('row');

  <>
    <Box display="flex" alignItems="center" mb={1}>
      <Box display="flex" alignItems="center" mx={1}>

        Direction:
      </Box>

          row

        </RadioGroup>
      </Box>
    </Box>
    <Carousel distance={500} offset={offset} direction={direction}>
      <Box style={{ [direction === 'row' ? 'height' : 'width']: 500, background: 'rgb(200, 0, 0)' }}>1</Box>
      <Box style={{ [direction === 'row' ? 'height' : 'width']: 500, background: 'rgb(0, 200, 0)' }}>2</Box>
