    <JsonTable
      data={Array(5).fill(0).map((_, index) => {
        return {
          id: index,
          name: `Smith ${index}`,
          surname: `Joe ${index}`,
          birthYear: 1979,
          birthCity: 77,
          status: Math.floor(Math.random() * 3) - 1,
        };
      })}
      dataSchema={{
        name: { title: 'Name' },
        surname: { title: 'Surname' },
        birthYear: { title: 'Birth Year' },
        birthCity: { title: 'Birth Place' },
        status: {
          title: 'Audit',
          enum: [-1, 1, 0],
          ui: {
            enum: ['Pending', 'Accepted', 'Rejected'],
            component: ({ children, rowValue, rowIndex, scope }) => {
              return scope === 'body' ? (
                <TableCell>
                  <Chip
                    label={children}
                    variant="ountlined"
                    color={
                      ['default', 'primary', 'secondary'][rowValue.status + 1]
                    }
                  />
                </TableCell>
              ) : (
                <TableCell>{children}</TableCell>
              );
            }
          },
        },
      }}
    />
