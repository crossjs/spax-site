    const [data, setData] = React.useState(Array(5).fill(0).map((_, index) => {
      return {
        id: index,
        name: `Smith ${index}`,
        surname: `Joe ${index}`,
        birthYear: 1979,
        birthCity: 77,
        status: Math.floor(Math.random() * 3) - 1,
        selected: Math.random() > 0.5,
      };
    }));
    const dataSchema = React.useMemo(() => {
      return {
        name: { title: 'Name' },
        surname: { title: 'Surname' },
        birthYear: { title: 'Birth Year' },
        birthCity: {
          title: 'Birth Place',
          map: { 76: 'İstanbul', 77: 'Şanlıurfa' },
        },
        status: {
          title: 'Audit',
          enum: [-1, 1, 0],
          ui: {
            enum: ['Pending', 'Accepted', 'Rejected'],
          },
        },
        actions: {
          title: 'Actions',
          type: 'null',
          ui: {
            component: ({ children, rowValue, rowIndex, scope }) => {
              return scope === 'body' ? (
                <TableCell>
                  <Button
                    variant="contained"
                    color="danger"
                    onClick={() => {
                      setConfirm([
                        JSON.stringify(rowValue, null, 2),
                        () => {
                          setData((v) => v.filter((_, idx) => idx !== rowIndex))
                        }
                      ]);
                    }}
                  >
                    Delete
                  </Button>
                  <Button variant="contained">Download</Button>
                </TableCell>
              ) : (
                <TableCell>{children}</TableCell>
              );
            }
          }
        }
      };
    })
    const [[message, callback], setConfirm] = React.useState([]);
    const handleCancel = React.useCallback(() => {
      setConfirm([]);
    }, []);
    const handleConfirm = React.useCallback(() => {
      callback();
      setConfirm([]);
    }, [callback]);
    return (
      <>
        <JsonTable
          data={data}
          dataSchema={dataSchema}
        />
        {message ? (
          <Confirm
            title="Confirm"
            onCancel={handleCancel}
            onConfirm={handleConfirm}
          >
            Are you sure you want to delete
            <pre>{message}</pre>
          </Confirm>
        ) : null}
      </>
    );
