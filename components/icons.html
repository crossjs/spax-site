const [iconList, setIconList] = React.useState(icons);
const [iconSize, setIconSize] = React.useState('default');
const [keyword, setKeyword] = React.useState('');
const onChange = React.useCallback((e) => {
  setKeyword(e.target.value.replace(/[^0-9a-z]/ig, '').trim().toLowerCase());
}, []);
const highlight = React.useCallback((name) => {
  if (keyword) {
    const matched = name.match(new RegExp(`(.*)(${keyword})(.*)`, 'i'));
    if (matched) {
      const [, a, b, c] = matched;
      return <>{a}<b style={{background: 'yellow'}}>{b}</b>{c}</>;
    }
  }
  return name;
}, [keyword]);
React.useEffect(() => {
  setIconList(Object.entries(icons)
  .reduce((o, [k, v]) => k.toLowerCase().indexOf(keyword) !== -1 ? ({
    ...o,
    [k]: v,
  }) : o, {}));
}, [keyword]);
return (
  <>
    <Grid container spacing={3} alignItems="center">
      <Grid item xs>
        <TextField
          fullWidth
          onChange={onChange}
          variant="outlined"
          InputProps={{
            startAdornment: (
              <InputAdornment position="start">
                <icons.Search />
              </InputAdornment>
            ),
          }}
        />
      </Grid>
      <Grid item xs>
        <RadioGroup row value={iconSize} onChange={(event) => {
          setIconSize(event.target.value);
        }}>
          <FormControlLabel value="default" control={<Radio />} label="Default" />
          <FormControlLabel value="small" control={<Radio />} label="Small" />
          <FormControlLabel value="large" control={<Radio />} label="Large" />
        </RadioGroup>
      </Grid>
    </Grid>
    <Box display="flex" flexWrap="wrap" mt={2}>
      {Object.keys(iconList).map((name) => {
        const Icon = iconList[name];
        return (
          <Box
            key={name}
            width={160}
            height={80}
            align="center"
            pt={2}
          >
            <Icon fontSize={iconSize} />
            <Box>{highlight(name)}</Box>
          </Box>
        );
      })}
    </Box>
  </>
);
