    Custom Fields

    classes.component classes.helperText

    i am description

    const CustomField = ({
      field: { name, value, onBlur, onChange },
      form: { errors },
      dataSchema: { description, readOnly },
      classes,
    }) => {
      const fieldError = errors[name];
      const helperText = fieldError || description;
      return readOnly ? (
        <Typography>{value || '\u00A0'}</Typography>
      ) : (
        <>
          <TextField
            // 
            className={classes.component}
            type="text"
            error={!!fieldError}
            name={name}
            value={value}
            onBlur={onBlur}
            onChange={onChange}
          />
          {helperText ? (
            <FormHelperText
              // 
              className={classes.helperText}
              error={!!fieldError}
            >{helperText}</FormHelperText>
          ) : null}
        </>
      );
    };
    return (
      <JsonForm
        layout="vertical"
        data={{}}
        dataSchema={{
          custom: {
            type: 'string',
            title: 'Custom',
            required: true,
            description: 'i am description',
            ui: {
              component: CustomField,
              placeholder: 'please input',
            },
          },
        }}
      />
    );
